\newcommand{\quickwordcount}[1]{%
  \immediate\write18{texcount -1 -sum -merge -q #1.tex > #1-words.sum }%

  \immediate\write18{texcount -1 -sum -merge -char -q #1.tex > #1-chars.sum }%

}
